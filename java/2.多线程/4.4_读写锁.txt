public class Cache {
	private Map<String, Object> cache = new HashMap<String, Object>();
	private ReadWriteLock rwl = new ReentrantReadWriteLock();
	//웨햢
	public Object getData(String key) {
		//웨
		rwl.readLock().lock();
		Object value = null;
		try {
			value = cache.get(key);
			if (value == null) {
				//햢浩썚턝퐁콨촮滎섉좛얢壕옾웵뿾햭
				rwl.readLock().unlock();
				rwl.writeLock().lock();
				try {
					if (value == null) {
						value = "get data from database";
					}
				} finally {
					rwl.writeLock().unlock();
				}
				//쟚웨
				rwl.readLock().lock();
			}
		} finally {
			rwl.readLock().unlock();
		}
		return value;
	}
}
